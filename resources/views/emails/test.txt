#define A7670E_TX    17

const char* ssid = "TestingWifi";

HardwareSerial a7670e(1);
Servo myServo;

// ===== CONFIGURATION =====
String phoneNumber = "+639070653250";

// ===== LOGIC VARIABLES =====
bool triggered = false;
bool pumpRunning = false;
bool servoRunning = false;
bool smsSent = false;
unsigned long pumpStartTime = 0;
unsigned long servoStartTime = 0;
const unsigned long PUMP_DURATION = 10000;

// ===== LOCATION & TIME VARIABLES =====
String latStr = "";
String lngStr = "";
String timeStr = "";
bool ntpSynced = false;

// ===== GET PHILIPPINE TIME FROM NTP =====
String getPhilippineTime() {
  struct tm timeinfo;
  
  if (!getLocalTime(&timeinfo, 1000)) {
    Serial.println("‚ùå Time not available");
    return "";
  }
  
  // Check if time is reasonable
  if (timeinfo.tm_year + 1900 < 2020) {
    Serial.println("‚ùå Time not synced yet");
    return "";
  }
  
  char buf[50];
  strftime(buf, sizeof(buf), "%b %d, %Y %H:%M:%S PHT", &timeinfo);
  
  Serial.println("‚úÖ Current time: " + String(buf));
  return String(buf);
}

// ===== HELPER FUNCTION =====
String waitForResponse(int timeout = 2000) {
  String response = "";
  unsigned long start = millis();
  while (millis() - start < timeout) {
    while (a7670e.available()) {
      char c = a7670e.read();
      response += c;
      Serial.write(c);
    }
  }
  Serial.println();
  return response;
}

// ===== FORMAT GPS TIME TO PHILIPPINE TIME =====
String formatGPSTime(String utc) {
  if (utc.length() < 14) return "";
  
  String year = utc.substring(0, 4);
  String month = utc.substring(4, 6);
  String day = utc.substring(6, 8);
  String hour = utc.substring(8, 10);
  String minute = utc.substring(10, 12);
  String second = utc.substring(12, 14);
  
  int hr = hour.toInt() + 8;
  int dy = day.toInt();
  
  if (hr >= 24) {
    hr -= 24;
    dy++;
  }
  
  String months[] = {"Jan", "Feb", "Mar", "Apr", "May", "Jun", 
                     "Jul", "Aug", "Sep", "Oct", "Nov", "Dec"};
  int mn = month.toInt() - 1;
  String monthName = (mn >= 0 && mn < 12) ? months[mn] : month;
  
  char buf[50];
  sprintf(buf, "%s %02d, %s %02d:%s:%s PHT", 
          monthName.c_str(), dy, year.c_str(), hr, minute.c_str(), second.c_str());
  
  return String(buf);
}

// ===== POWER ON MODULE =====
void powerOnA7670E() {
  Serial.println("üîå Powering on A7670E...");
  pinMode(A7670E_PWR, OUTPUT);
  digitalWrite(A7670E_PWR, HIGH);
  delay(500);
  digitalWrite(A7670E_PWR, LOW);
  delay(1000);
  digitalWrite(A7670E_PWR, HIGH);
  delay(5000);
}

// ===== TRY GPS MODULE (QUICK - 10 SECONDS MAX) =====
bool tryGPSModule() {
  Serial.println("\nüõ∞Ô∏è Trying GPS module (10 sec)...");
  
  a7670e.println("AT+CGNSPWR=1");
  String r = waitForResponse(2000);
  
  if (r.indexOf("ERROR") >= 0) {
    Serial.println("‚ùå GPS module not available");
    return false;
  }
  
  for (int i = 0; i < 5; i++) {
    Serial.printf("GPS attempt %d/5...\n", i + 1);
    
    a7670e.println("AT+CGNSINF");
    String resp = waitForResponse(2000);
    
    if (resp.indexOf("+CGNSINF:") >= 0) {
      int p = resp.indexOf("+CGNSINF:");
      String data = resp.substring(p + 10);
      
      int c1 = data.indexOf(',');
      int c2 = data.indexOf(',', c1 + 1);
      int c3 = data.indexOf(',', c2 + 1);
      int c4 = data.indexOf(',', c3 + 1);
      int c5 = data.indexOf(',', c4 + 1);
      
      if (c5 > 0) {
        String fix = data.substring(c1 + 1, c2);
        String utc = data.substring(c2 + 1, c3);
        String lat = data.substring(c3 + 1, c4);
        String lon = data.substring(c4 + 1, c5);
        
        fix.trim();
        utc.trim();
        lat.trim();
        lon.trim();
        
        if (fix == "1" && lat != "0.0" && lon != "0.0" && lat.length() > 0) {
          latStr = lat;
          lngStr = lon;
          
          String gpsTime = formatGPSTime(utc);
          if (gpsTime != "") {
            timeStr = gpsTime;
            Serial.println("‚úÖ GPS TIME: " + timeStr);
          }
          
          Serial.println("‚úÖ GPS FIX!");
          Serial.println("Lat: " + latStr);
          Serial.println("Lng: " + lngStr);
          return true;
        }
      }
    }
    
    delay(1000);
  }
  
  Serial.println("‚ö†Ô∏è No GPS fix");
  return false;
}

// ===== GET LOCATION FROM IP GEOLOCATION API =====
bool getLocationFromIP() {
  if (WiFi.status() != WL_CONNECTED) {
    Serial.println("‚ùå WiFi not connected");
    return false;
  }
  
  Serial.println("\nüåê Getting location from IP geolocation...");
  
  HTTPClient http;
  http.begin("http://ip-api.com/json/?fields=lat,lon");
  
  int httpCode = http.GET();
  
  if (httpCode == 200) {
    String payload = http.getString();
    Serial.println("Response: " + payload);
    
    DynamicJsonDocument doc(1024);
    DeserializationError error = deserializeJson(doc, payload);
    
    if (!error) {
      float lat = doc["lat"];
      float lon = doc["lon"];
      
      latStr = String(lat, 6);
      lngStr = String(lon, 6);
      
      Serial.println("‚úÖ IP Geolocation acquired!");
      Serial.println("Lat: " + latStr);
      Serial.println("Lng: " + lngStr);
      
      http.end();
      return true;
    } else {
      Serial.println("‚ùå JSON parse error");
    }
  } else {
    Serial.print("‚ùå HTTP error: ");
    Serial.println(httpCode);
  }
  
  http.end();
  return false;
}

// ===== SEND SMS =====
void sendSMS(String msg) {
  Serial.println("\n‚û°Ô∏è Sending SMS...");
  
  while (a7670e.available()) a7670e.read();
  delay(500);
  
  a7670e.println("AT");
  waitForResponse(1000);
  
  a7670e.println("AT+CMGF=1");
  waitForResponse(1000);
  
  a7670e.println("AT+CSCS=\"GSM\"");
  waitForResponse(1000);
  
  while (a7670e.available()) a7670e.read();
  
  a7670e.print("AT+CMGS=\"");
  a7670e.print(phoneNumber);
  a7670e.println("\"");
  
  delay(2000);
  
  bool gotPrompt = false;
  unsigned long start = millis();
  while (millis() - start < 5000) {
    if (a7670e.available()) {
      char c = a7670e.read();
      Serial.write(c);
      if (c == '>') {
        gotPrompt = true;
        Serial.println("\n‚úÖ Prompt");
        break;
      }
    }
  }
  
  if (!gotPrompt) {
    Serial.println("‚ùå No prompt!");
    a7670e.write(27);
    return;
  }
  
  delay(500);
  a7670e.print(msg);
  delay(500);
  a7670e.write(26);
  
  Serial.println("üì§ Waiting...");
  String resp = "";
  start = millis();
  while (millis() - start < 30000) {
    while (a7670e.available()) {
      char c = a7670e.read();
      resp += c;
      Serial.write(c);
    }
    if (resp.indexOf("+CMGS:") >= 0) {
      Serial.println("\n‚úÖ SMS SENT!");
      return;
    }
    if (resp.indexOf("ERROR") >= 0) {
      Serial.println("\n‚ùå FAILED!");
      return;
    }
  }
  Serial.println("\n‚ö†Ô∏è Timeout");
}

// ===== SETUP =====
void setup() {
  pinMode(WATER_SENSOR, INPUT);
  pinMode(RELAY_PIN, OUTPUT);
  digitalWrite(RELAY_PIN, HIGH);
  
  Serial.begin(115200);
  delay(1000);
  
  Serial.println("\n========================================");
  Serial.println("  FLOOD DETECTION SYSTEM");
  Serial.println("  GPS ‚Üí WiFi Fallback");
  Serial.println("========================================\n");
  
  // Connect to WiFi
  Serial.println("üì∂ Connecting to WiFi...");
  WiFi.mode(WIFI_STA);
  WiFi.begin(ssid, password);
  
  int attempts = 0;
  while (WiFi.status() != WL_CONNECTED && attempts < 20) {
    delay(500);
    Serial.print(".");
    attempts++;
  }
  Serial.println();
  
  if (WiFi.status() == WL_CONNECTED) {
    Serial.println("‚úÖ WiFi Connected!");
    Serial.print("IP: ");
    Serial.println(WiFi.localIP());
    
    // Sync NTP time immediately at startup
    Serial.println("\nüïê Syncing NTP time...");
    configTime(8 * 3600, 0, "pool.ntp.org", "time.nist.gov");
    
    // Wait for time sync
    struct tm timeinfo;
    int syncAttempts = 0;
    while (syncAttempts < 15) {
      if (getLocalTime(&timeinfo, 1000)) {
        if (timeinfo.tm_year + 1900 > 2020) {
          ntpSynced = true;
          char buf[50];
          strftime(buf, sizeof(buf), "%b %d, %Y %H:%M:%S PHT", &timeinfo);
          Serial.println("‚úÖ NTP synced: " + String(buf));
          break;
        }
      }
      Serial.print(".");
      delay(1000);
      syncAttempts++;
    }
    Serial.println();
    
    if (!ntpSynced) {
      Serial.println("‚ö†Ô∏è NTP sync timeout");
    }
  } else {
    Serial.println("‚ö†Ô∏è WiFi failed - IP geolocation & NTP won't work\n");
  }
  
  powerOnA7670E();
  
  Serial.println("üì° Initializing A7670E...");
  a7670e.begin(115200, SERIAL_8N1, A7670E_RX, A7670E_TX);
  delay(2000);
  
  a7670e.println("AT");
  waitForResponse(2000);
  
  a7670e.println("ATE0");
  waitForResponse(1000);
  
  Serial.println("\n‚è≥ Network registration...");
  for (int i = 0; i < 40; i++) {
    a7670e.println("AT+CREG?");
    String r = waitForResponse(1000);
    if (r.indexOf("+CREG: 0,1") >= 0 || r.indexOf("+CREG: 0,5") >= 0) {
      Serial.println("‚úÖ Network OK!");
      break;
    }
    if (i % 5 == 0) Serial.printf("Searching %d/40\n", i);
    delay(1000);
  }
  
  a7670e.println("AT+CMGF=1");
  waitForResponse(1000);
  
  // Enable GPS module in background
  a7670e.println("AT+CGNSPWR=1");
  waitForResponse(2000);
  
  Serial.println("\nüîß Servo...");
  myServo.setPeriodHertz(50);
  myServo.attach(SERVO_PIN, 500, 2400);
  myServo.write(0);
  
  Serial.println("\n========================================");
  Serial.println("‚úÖ READY!");
  Serial.println("========================================");
  Serial.println("Time Priority: GPS ‚Üí WiFi NTP");
  Serial.println("Location Priority: GPS ‚Üí IP Geolocation");
  Serial.println("Monitoring for water...\n");
}

// ===== LOOP =====
void loop() {
  int waterDetected = digitalRead(WATER_SENSOR);
  
  static unsigned long lastPrint = 0;
  if (!triggered && millis() - lastPrint > 10000) {
    Serial.print("üíß ");
    Serial.println(waterDetected == HIGH ? "DETECTED" : "Normal");
    lastPrint = millis();
  }
  
  if (waterDetected == HIGH && !triggered) {
    Serial.println("\nüö® WATER DETECTED!");
    triggered = true;
    pumpRunning = true;
    pumpStartTime = millis();
    
    digitalWrite(RELAY_PIN, LOW);
    Serial.println("üîå Pump ON\n");
  }
  
  if (pumpRunning && millis() - pumpStartTime >= PUMP_DURATION) {
    pumpRunning = false;
    servoRunning = true;
    servoStartTime = millis();
    
    digitalWrite(RELAY_PIN, HIGH);
    Serial.println("üîå Pump OFF");
    Serial.println("üîÑ Servo...\n");
    myServo.write(0);
  }
  
  if (servoRunning) {
    unsigned long e = millis() - servoStartTime;
    
    if (e < 2000) {
      myServo.write(map(e, 0, 2000, 0, 180));
    } else {
      servoRunning = false;
      myServo.write(0);
      Serial.println("‚úÖ Servo done\n");
      
      // ===== LOCATION PRIORITY =====
      bool hasLocation = tryGPSModule();
      
      if (!hasLocation) {
        Serial.println("GPS failed, trying IP geolocation...");
        hasLocation = getLocationFromIP();
      }
      
      // ===== TIME PRIORITY =====
      // PRIORITY 1: GPS time (already set in tryGPSModule if successful)
      // PRIORITY 2: WiFi NTP time (synced at startup)
      if (timeStr == "" && ntpSynced) {
        Serial.println("GPS time not available, using NTP...");
        timeStr = getPhilippineTime();
      }
      
      // PRIORITY 3: Final fallback
      if (timeStr == "") {
        timeStr = "Time unavailable";
        Serial.println("‚ö†Ô∏è All time sources failed");
      }
      
      // Build SMS
      String sms = "FLOOD ALERT!\n";
      sms += "Harness sensor detected water\n";
      sms += "Inflation: 10s\n";
      sms += "Status: UNSAFE\n";
      sms += "Time: " + timeStr + "\n";
      
      if (hasLocation && latStr != "" && lngStr != "") {
        sms += "Location: https://www.google.com/maps?q=" + latStr + "," + lngStr;
      } else {
        sms += "Location: Unavailable";
      }
      
      Serial.println("üì± SMS:");
      Serial.println("‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ");
      Serial.println(sms);
      Serial.println("‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ‚îÄ\n");
      
      sendSMS(sms);
      
      smsSent = true;
      Serial.println("‚úÖ COMPLETE!\n");
    }
  }
  
  if (triggered && smsSent && waterDetected == LOW) {
    triggered = false;
    pumpRunning = false;
    servoRunning = false;
    smsSent = false;
    latStr = "";
    lngStr = "";
    timeStr = "";
    
    Serial.println("üíö Reset\n");
  }
  
  delay(50);
}
